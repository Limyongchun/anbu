4. 카드 데이터 구조

type Parent = {
  id: string
  name: string
  profileImage: string

  status: "safe" | "warning" | "danger"

  locationLabel: string // "집", "노인정", "이동중"
  lastActivity: string // "15분 전 사진 보기"
  lastUpdateTime: string // "오후 5:42"

  battery?: number
}


⸻

5. 카드 정렬 (핵심)

const priority = {
  danger: 0,
  warning: 1,
  safe: 2
}

const sortedParents = parents.sort(
  (a, b) => priority[a.status] - priority[b.status]
)

7. 상태 문구 함수 (핵심 UX)

function getLocationText(data: Parent) {
  if (data.status === "danger") {
    return "위치 확인이 필요해요"
  }

  if (data.locationLabel === "이동중") {
    return "지금 이동 중이세요"
  }

  return `지금 ${data.locationLabel}에 계세요`
}


⸻

8. 상태별 스타일

function getStatusStyle(status: string) {
  switch (status) {
    case "safe":
      return { borderLeftColor: "#4CAF50" }
    case "warning":
      return { borderLeftColor: "#FFC107" }
    case "danger":
      return { borderLeftColor: "#F44336" }
  }
}
